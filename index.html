async function loadDonations() { try { const response = await fetch("https://script.google.com/macros/s/AKfycby3AtgmX1gCDd2KUeYT8TkUBkVKoh_VH0q3oflLp6Bp3tse0SOETZgGV9wgiP8fT4FY/exec?action=getDonations"); if (!response.ok) throw new Error("データ取得エラー"); const donations = await response.json(); console.log("レスポンスデータ:", donations); if (!Array.isArray(donations)) throw new Error("データ形式エラー"); // 最新の投稿を上に表示 const list = donations.map(donation => `

${donation.donor} 様

品物: ${donation.itemName}

メッセージ: ${donation.message}

寄付日時: ${donation.timestamp}

ニックネーム: ${donation.nickname}

${donation.imageUrl ? `` : ''}
`).join(''); document.getElementById("donation-list").innerHTML = list; } catch (error) { console.error("データ表示エラー:", error); document.getElementById("donation-list").innerHTML = "データを取得できませんでした。"; } } loadDonations();
